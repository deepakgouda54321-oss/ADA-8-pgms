#include<stdio.h>
void main(){
    int d[10][10],i,j,k,n;
    printf("\nWarshall's Algorithm\n");
    printf("Enter the number of vertices:");
    scanf("%d",&n);
    printf("Enter the adjecency matrix:\n");
    for(i=0;i<n;i++)
    for(j=0;j<n;j++)
    scanf("%d",&d[i][j]);
    for(k=0;k<n;k++)
    for(i=0;i<n;i++)
    for(j=0;j<n;j++)
    if(d[i][k]==1 && d[k][j]==1)
    d[i][j]=1;
    printf("Transitive closure matrix:\n");
    for(i=0;i<n;i++){
        for(j=0;j<n;j++)
        printf("%d\t",d[i][j]);
        printf("\n");
    }
}
