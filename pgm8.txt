#include<stdio.h>
int w[10],x[10],n,d;
void subset(int k,int sum){
    if(sum==d){
        for(int i=0;i<=k;i++)
        if(x[i])
        printf("%d",w[i]);
        printf("\n");
        return;
    }
    if(k==n || sum>d)
    return;
    x[k]=1;
    subset(k+1,sum+w[k]);
    x[k]=0;
    subset(k+1,sum);
}
int main(){
    int i;
    printf("Enter n and target:");
    scanf("%d%d",&n,&d);
    printf("Enter elements:\n");
    for(i=0;i<n;i++)
    scanf("%d",&w[i]);
    printf("Subsets are:\n");
    subset(0,0);
    return 0;
}
