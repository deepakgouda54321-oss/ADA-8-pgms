#include<stdio.h>
int visited[10],mincost;

void main(){
    int n,i,j,count=1,u,v,cost[10][10],min;
    printf("Prims Algorithm\n");
    printf("Enter the number of nodes:");
    scanf("%d",&n);
    printf("Enter the adjecency matrix:\n");
    for(i=0;i<n;i++)
    for(j=0;j<n;j++)
    scanf("%d",&cost[i][j]);
    visited[0]=1;
    printf("the edges of minimum cost spanning tree are:\n");
    while(count<n){
        min=99;
        for(i=0;i<n;i++)
        for(j=0;j<n;j++)
        if(cost[i][j]<min && visited[i]==1 && visited[j]==0){
            min=cost[i][j];
            u=i;
            v=j;
        }
        printf("%d edge (%d,%d)=%d\n",count++,u,v,min);
        mincost=mincost+min;
        visited[v]=1;
    }
    printf("\nMinimum cost:%d\n",mincost);
}
