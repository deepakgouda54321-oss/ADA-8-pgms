#include<stdio.h>
int main() {
    int n,cost[10][10],dist[10],visit[10]={0};
    int i,j,src,u,min;
    printf("Enter number of nodes: ");
    scanf("%d",&n);
    printf("Enter cost matrix:\n");
    for(i=0;i<n;i++)
        for(j=0;j<n;j++)
            scanf("%d",&cost[i][j]);
    printf("Enter source node: ");
    scanf("%d",&src);
    for(i=0;i<n;i++)
    dist[i]=99;
    dist[src]=0;
    for(i=0;i<n;i++) {
        min=99;
        for(j=0;j<n;j++)
            if(!visit[j] && dist[j]<min)
                min=dist[u=j];
        visit[u]=1;
        for(j=0;j<n;j++)
            if(!visit[j] && dist[u]+cost[u][j]<dist[j])
                dist[j]=dist[u]+cost[u][j];
    }
    printf("Shortest distances:\n");
    for(i=0;i<n;i++)
        printf("Node %d = %d\n",i,dist[i]);
    return 0;
}
