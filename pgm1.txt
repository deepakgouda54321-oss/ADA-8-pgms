#include<stdio.h>
#define MAX 20
#define INF 999
int p[MAX];
int find(int x){
    while(p[x])
    x=p[x];
    return x;
}
void uni(int a,int b){
    p[find(b)]=find(a);
}
int main(){
    int n,c[MAX][MAX],i,j,k,min,u,v,cost=0;
    printf("Enter the no of vertices:");
    scanf("%d",&n);
    printf("Enter the cost matrix:\n");
    for(i=1;i<=n;i++)
    for(j=1;j<=n;j++)
    scanf("%d",&c[i][j]);
    for(k=1;k<n;k++){
        min=INF;
        for(i=1;i<=n;i++)
        for(j=1;j<=n;j++)
        if(c[i][j]<min && find(i)!=find(j)){
            min=c[i][j];
            u=i;
            v=j;
        }
        uni(u,v);
        cost+=min;
        printf("%d ->%d\n",u,v);
        c[u][v]=c[v][u]=INF;
    }
printf("Minimum cost=%d",cost);
return 0;
}
